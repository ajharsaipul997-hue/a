constructor(x,y,color,name="Player") {
this.x = x;
this.y = y;
this.color = color;
this.radius = 15;
this.speed = 3;
this.sprinting = false;
this.name = name;
}
draw() {
ctx.fillStyle = this.color;
ctx.beginPath();
// Ellipse untuk efek sprite miring 3D sederhana
ctx.ellipse(this.x, this.y, this.radius*1.2, this.radius, 0, 0, 2*Math.PI);
ctx.fill();
ctx.strokeStyle = 'rgba(0,0,0,0.3)';
ctx.lineWidth = 4;
ctx.stroke();
}
move(dx,dy) {
let spd = this.sprinting ? this.speed * 2 : this.speed;
this.x += dx * spd;
this.y += dy * spd;
// Batasi lapangan
this.x = Math.min(fieldWidth - this.radius, Math.max(this.radius, this.x));
this.y = Math.min(fieldHeight - this.radius, Math.max(this.radius, this.y));
}
}
